1   package abandonallhope.events.action;
2   
3   import abandonallhope.domain.Inventory;
4   import abandonallhope.domain.Survivor;
5   import abandonallhope.domain.weapons.Weapon;
6   import abandonallhope.logic.ResourceEvents;
7   import javafx.beans.value.ChangeListener;
8   import javafx.beans.value.ObservableValue;
9   
10  /**
11   * Event for giving a survivor a new weapon.
12   *
13   * @author kipsu
14   */
15  public class WeaponEvent implements ChangeListener<String> {
16  
17  	private Inventory inventory;
18  	private Survivor survivor;
19  	private ResourceEvents resEvent;
20  
21  	/**
22  	 * Creates a new weapon event handling changing survivor weapons.
23  	 *
24  	 * @param inventory
25  	 * @param survivor
26  	 * @param resEvent
27  	 */
28  	public WeaponEvent(Inventory inventory, Survivor survivor, ResourceEvents resEvent) {
29  		this.inventory = inventory;
30  		this.survivor = survivor;
31  		this.resEvent = resEvent;
32  	}
33  
34  	@Override
35  	public void changed(ObservableValue<? extends String> ov, String previousWeapon, String newWeapon) {
36  		if (newWeapon.equals("none")) {
37  			removeSurvivorWeapon();
38  		} else {
39  			giveSurvivorAWeapon();
40  		}
41  	}
42  
43  	private void removeSurvivorWeapon() {
44  		Weapon weapon = survivor.getWeapon();
45  		setAvailableForOtherSurvivors(weapon);
46  		inventory.addWeapons(weapon);
47  		survivor.setWeapon(null);
48  	}
49  
50  	private void setAvailableForOtherSurvivors(Weapon weapon) {
51  		if (!inventory.containsWeapon(weapon)) {
52  			resEvent.triggerNewWeaponEvent(weapon);
53  		}
54  	}
55  
56  	private void giveSurvivorAWeapon() {
57  		Weapon weapon = getWeaponFromInventory();
58  		survivor.setWeapon(weapon);
59  		setUnavailableForOtherSurvivors(weapon);
60  	}
61  
62  	private Weapon getWeaponFromInventory() {
63  		Weapon weapon = inventory.getWeapons().get(0);
64  		inventory.getWeapons().remove(0);
65  		return weapon;
66  	}
67  
68  	private void setUnavailableForOtherSurvivors(Weapon weapon) {
69  		if (inventory.getWeapons().isEmpty()) {
70  			resEvent.triggerDeleteWeaponEvent(weapon);
71  		}
72  	}
73  
74  }
