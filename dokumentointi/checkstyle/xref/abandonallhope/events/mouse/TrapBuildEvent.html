2   package abandonallhope.events.mouse;
3   
4   import abandonallhope.domain.Point;
5   import abandonallhope.domain.constructions.Trap;
6   import abandonallhope.logic.Game;
7   import abandonallhope.ui.GameCanvas;
8   import javafx.event.EventHandler;
9   import javafx.scene.input.MouseEvent;
10  
11  /**
12   * Event handler to build traps once user clicks the build location.
13   * @author kipsu
14   */
15  public class TrapBuildEvent implements EventHandler<MouseEvent> {
16  
17  	private Game game;
18  	private GameCanvas canvas;
19  	private Trap trap;
20  
21  	/**
22  	 * Build event will build the trap the player is hovering over game field
23  	 * on click. Will also remove trap building related event listeners.
24  	 * @param game game where the trap will be built.
25  	 * @param canvas canvas containing the event listeners.
26  	 * @param trap Trap type.
27  	 */
28  	public TrapBuildEvent(Game game, GameCanvas canvas, Trap trap) {
29  		this.game = game;
30  		this.canvas = canvas;
31  		this.trap = trap;
32  	}
33  
34  	@Override
35  	public void handle(MouseEvent t) {
36  		trap.setLocation(new Point(t.getX(), t.getY()));
37  		game.add(trap);
38  		game.getInventory().payResources(trap.getCost());
39  		canvas.removeTrapBuildingEventListeners();
40  	}
41  	
42  }
