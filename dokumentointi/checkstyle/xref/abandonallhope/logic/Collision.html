1   package abandonallhope.logic;
2   
3   import abandonallhope.domain.MovingObject;
4   import abandonallhope.domain.Point;
5   import java.util.List;
6   
7   /**
8    * Class containing static methods to check if two objects collide or which is
9    * the nearest by other object
10   *
11   * @author kipsu
12   */
13  public class Collision {
14  
15  	/**
16  	 * Checks if a zombie is colliding with at least one survivor and returns
17  	 * the survivor if so.
18  	 *
19  	 * @param source whose collisions are checked
20  	 * @param targets targets whose collisions are checked
21  	 * @return target who is being collided with, other wise null
22  	 */
23  	public static MovingObject hitTest(MovingObject source, List<? extends MovingObject> targets) {
24  		for (MovingObject target : targets) {
25  			if (source.occupiedArea().intersects(target.occupiedArea())) {
26  				return target;
27  			}
28  		}
29  		return null;
30  	}
31  
32  	/**
33  	 * Return the point containing the nearest zombie or survivor.
34  	 *
35  	 * @param person Zombie or a survivor whose vicinity is checked
36  	 * @param enemies list of zombies or survivors among which the point is
37  	 * searched from
38  	 * @return Point object for the nearest found person
39  	 */
40  	public static Point nearestPersonLocation(MovingObject person, List<? extends MovingObject> enemies) {
41  		Point nearest = nearestPerson(person, enemies).getLocation();
42  		return nearest;
43  	}
44  
45  	/**
46  	 * Return the nearest zombie or survivor
47  	 *
48  	 * @param person Zombie or a survivor whose vicinity is checked
49  	 * @param enemies list of zombies or survivors among which the point is
50  	 * searched from
51  	 * @return The nearest found person
52  	 */
53  	public static MovingObject nearestPerson(MovingObject person, List<? extends MovingObject> enemies) {
54  		try {
55  			MovingObject nearest = enemies.get(0);
56  			return nearest(enemies, person, nearest);
57  		} catch (Exception e) {
58  			return null;
59  		}
60  	}
61  
62  	/**
63  	 * Get the absolute distance between two points. Point order has no
64  	 * significance.
65  	 *
66  	 * @param x Point object x
67  	 * @param y Point object y
68  	 * @return Absolute distance between the points as double
69  	 */
70  	public static double distanceBetween(Point x, Point y) {
71  		return Math.abs(y.x - x.x) + Math.abs(y.y - x.y);
72  	}
73  
74  	private static MovingObject nearest(List<? extends MovingObject> enemies, MovingObject person, MovingObject nearest) {
75  		double nearesDifference = Double.MAX_VALUE;
76  		for (MovingObject enemy : enemies) {
77  			if (enemyIsTrapped(enemy)) {
78  				continue;
79  			}
80  			double difference = distanceBetween(person.getLocation(), enemy.getLocation());
81  			if (nearest.isTrapped() || difference < nearesDifference) {
82  				nearest = enemy;
83  				nearesDifference = difference;
84  			}
85  		}
86  		return nearest.isTrapped() ? null : nearest;
87  	}
88  
89  	private static boolean enemyIsTrapped(MovingObject enemy) {
90  		if (enemy.isTrapped()) {
91  			return true;
92  		}
93  		return false;
94  	}
95  
96  }
