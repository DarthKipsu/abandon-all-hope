1   package abandonallhope.ui;
2   
3   import abandonallhope.domain.constructions.TrapType;
4   import abandonallhope.domain.constructions.WallType;
5   import abandonallhope.events.action.DeselectEvent;
6   import abandonallhope.events.action.TrapEvent;
7   import abandonallhope.events.action.WallEvent;
8   import abandonallhope.logic.Game;
9   import javafx.geometry.Insets;
10  import javafx.scene.control.Button;
11  import javafx.scene.control.Label;
12  import javafx.scene.layout.VBox;
13  import javafx.scene.text.Font;
14  import javafx.scene.text.FontWeight;
15  
16  /**
17   * Right ui panel containing objects about building and weapons.
18   *
19   * @author kipsu
20   */
21  public class BuildPanel {
22  
23  	private VBox vbox;
24  	private Game game;
25  	private GameCanvas canvas;
26  
27  	/**
28  	 * Creates a new building and weapons panel, displaying building types and
29  	 * available weapons.
30  	 */
31  	public BuildPanel(Game game, GameCanvas canvas) {
32  		vbox = new VBox();
33  		vbox.setPadding(new Insets(10));
34  		vbox.setSpacing(8);
35  		this.game = game;
36  		this.canvas = canvas;
37  		createVBoxContent();
38  	}
39  
40  	public VBox getVbox() {
41  		return vbox;
42  	}
43  
44  	private void createVBoxContent() {
45  		addDeselectionButton();
46  		
47  		addTitle("Build a wall:");
48  		addWallButton("Wooden", WallType.WOODEN);
49  		
50  		addTitle("Build a trap:");
51  		addTrapButton("Bear Iron", TrapType.BEARIRON);
52  		addTrapButton("Pit", TrapType.PIT);
53  	}
54  	
55  	private void addDeselectionButton() {
56  		Button deselect = new Button("Uselect all");
57  		deselect.setPrefSize(100, 20);
58  		deselect.setOnAction(new DeselectEvent(canvas, game));
59  		vbox.getChildren().add(deselect);
60  	}
61  
62  	private void addTitle(String text) {
63  		Label title = new Label(text);
64  		title.setFont(Font.font("Arial", FontWeight.BOLD, 14));
65  		vbox.getChildren().add(title);
66  	}
67  
68  	private void addWallButton(String name, WallType wallType) {
69  		Button wall = new Button(name);
70  		wall.setPrefSize(100, 20);
71  		wall.setOnAction(new WallEvent(canvas, game, wallType));
72  		vbox.getChildren().add(wall);
73  	}
74  
75  	private void addTrapButton(String name, TrapType trapType) {
76  		Button trap = new Button(name);
77  		trap.setPrefSize(100, 20);
78  		trap.setOnAction(new TrapEvent(canvas, game, trapType));
79  		vbox.getChildren().add(trap);
80  	}
81  }
