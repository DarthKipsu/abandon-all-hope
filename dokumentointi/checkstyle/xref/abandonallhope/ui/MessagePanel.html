2   package abandonallhope.ui;
3   
4   import abandonallhope.logic.Game;
5   import javafx.geometry.Insets;
6   import javafx.scene.layout.VBox;
7   import javafx.scene.text.Text;
8   
9   /**
10   * Panel showing the player messages containing info about what is happening in the game.
11   * @author kipsu
12   */
13  public class MessagePanel {
14  	
15  	private VBox vbox;
16  
17  	/**
18  	 * Creates a new message panel displaying messages to player.
19  	 */
20  	public MessagePanel(Game game) {
21  		vbox = new VBox();
22  		vbox.setPadding(new Insets(10));
23  		vbox.setSpacing(4);
24  		vbox.setPrefHeight(120);
25  		game.setMessages(this);
26  	}
27  
28  	public VBox getVbox() {
29  		return vbox;
30  	}
31  	
32  	/**
33  	 * Shows the player a new message and removes old ones from sight if message count
34  	 * is greater than 5.
35  	 * @param message message to be displayed
36  	 */
37  	public void addMessage(String message) {
38  		vbox.getChildren().add(0, new Text(message));
39  		if (vbox.getChildren().size() > 5) {
40  			vbox.getChildren().remove(5);
41  		}
42  	}
43  	
44  }
